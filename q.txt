# Path to your exported Qualys Policy XML
$xmlPath = "C:\Path\To\Your\qualys_policy_export.xml"
$outputCsv = "C:\Temp\qualys_registry_gpp.csv"

# Load XML
[xml]$xml = Get-Content $xmlPath
$results = @()

# Loop through each control
$controls = $xml.SelectNodes("//CONTROL")

foreach ($control in $controls) {
    $controlId = $control.ID.'#text'

    foreach ($tech in $control.SelectNodes(".//TECHNOLOGY")) {
        $techName = $tech.NAME.'#text'

        foreach ($ctrlNode in $tech.SelectNodes(".//CTRL")) {
            foreach ($dpNode in $ctrlNode.SelectNodes(".//DP")) {
                $kVal = $dpNode.K
                $vVal = $dpNode.V
                $desc = $dpNode.InnerText

                # Attempt to detect registry keys
                if ($desc -match "HKEY_") {
                    $regKey = $null
                    $valueName = $null
                    $expectedValue = $null
                    $valueType = "DWord"  # Default for GPP

                    # Extract registry key from description
                    if ($desc -match "HKEY_[A-Z_\\]+\\([a-zA-Z0-9]+)?") {
                        $regKey = ($desc -split "registry key location|key")[1] -replace "[:\[\]]", "" -replace "\s{2,}", " "
                        $regKey = ($regKey -split "value|must|that|if|from")[0].Trim()
                    }

                    if ($regKey -and $regKey -match "\\(.+?)$") {
                        $valueName = ($regKey -split "\\")[-1].Trim()
                    }

                    # Try V element for expected value
                    $expectedValue = $vVal.'#cdata-section' -replace "[^\x20-\x7E]", ""

                    # Determine value type (basic logic)
                    if ($expectedValue -match "^[0-9]+$") {
                        $valueType = "DWord"
                    } elseif ($expectedValue) {
                        $valueType = "String"
                    }

                    if ($regKey -and $valueName -and $expectedValue) {
                        $results += [pscustomobject]@{
                            "Control ID"     = $controlId
                            "Technology"     = $techName
                            "K Value"        = $kVal
                            "Registry Key"   = $regKey
                            "Value Name"     = $valueName
                            "Expected Value" = $expectedValue
                            "Value Type"     = $valueType
                        }
                    }
                }
            }
        }
    }
}

# Export for GPP import
if ($results.Count -gt 0) {
    $results | Export-Csv -Path $outputCsv -NoTypeInformation -Encoding UTF8
    Write-Host "✅ Exported $($results.Count) registry entries for GPP to: $outputCsv"
} else {
    Write-Warning "No registry entries found."
}
