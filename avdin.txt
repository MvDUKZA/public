# Parameters
$csvPath = "C:\Scripts\user_device_mapping.csv"
$logPath = "C:\Scripts\avd_intune_assignment.log"
$graphScope = "DeviceManagementManagedDevices.ReadWrite.All"

function Write-Log {
    param([string]$message)
    $timestamp = Get-Date -Format "yyyy-MM-dd HH:mm:ss"
    Add-Content -Path $logPath -Value "$timestamp - $message"
    Write-Output "$timestamp - $message"
}

# Connect to Azure
try {
    Connect-AzAccount -ErrorAction Stop
    Write-Log "Connected to Azure."
} catch {
    Write-Error "Failed to connect to Azure: $_"
    exit 1
}

# Connect to Graph
try {
    Import-Module Microsoft.Graph -ErrorAction Stop
    Connect-MgGraph -Scopes $graphScope -ErrorAction Stop
    Write-Log "Connected to Microsoft Graph with scope: $graphScope"
} catch {
    Write-Error "Failed to connect to Microsoft Graph: $_"
    exit 1
}

# Load CSV
if (-not (Test-Path $csvPath)) {
    Write-Error "CSV file not found: $csvPath"
    Disconnect-MgGraph
    exit 1
}

$records = Import-Csv -Path $csvPath
if (-not $records) {
    Write-Error "CSV is empty."
    Disconnect-MgGraph
    exit 1
}

foreach ($row in $records) {
    $userUPN = $row.UserUPN.Trim()
    $deviceName = $row.VDIName.Trim()
    $hostPool = $row.HostPoolName.Trim()
    $resourceGroup = $row.ResourceGroupName.Trim()

    Write-Log "Processing: $userUPN -> $deviceName [$hostPool / $resourceGroup]"

    # ===== Azure AVD Assignment =====
    try {
        $fqdn = "$deviceName.domain.local"  # Adjust if your session hosts are not FQDN
        $sessionHost = Get-AzWvdSessionHost -ResourceGroupName $resourceGroup -HostPoolName $hostPool `
            | Where-Object { $_.Name -like "*$deviceName*" }

        if (-not $sessionHost) {
            throw "Session host '$deviceName' not found in host pool '$hostPool'."
        }

        # Assign the user
        Update-AzWvdSessionHost -ResourceGroupName $resourceGroup -HostPoolName $hostPool `
            -Name $sessionHost.Name -AssignedUser $userUPN -ErrorAction Stop

        Write-Log "SUCCESS: Assigned AVD session host '$deviceName' to user '$userUPN'."
    } catch {
        Write-Log "ERROR: AVD assignment failed for '$userUPN' to '$deviceName'. $_"
        continue
    }

    # ===== Intune Assignment =====
    try {
        $device = Get-MgDeviceManagementManagedDevice -Filter "deviceName eq '$deviceName'" -ErrorAction Stop
        if (-not $device) { throw "Device '$deviceName' not found in Intune." }
        if ($device.ManagementState -ne "mdm") { throw "Device '$deviceName' not MDM-managed." }

        $deviceId = $device.Id
        $uri = "https://graph.microsoft.com/beta/deviceManagement/managedDevices/$deviceId/assignUserToDevice"
        $body = @{ userPrincipalName = $userUPN } | ConvertTo-Json -Depth 2
        $headers = @{ "Content-Type" = "application/json" }

        Invoke-MgGraphRequest -Method POST -Uri $uri -Headers $headers -Body $body -ErrorAction Stop
        Write-Log "SUCCESS: Assigned Intune primary user '$userUPN' to '$deviceName'."
    } catch {
        $errorDetails = $_.Exception.Response.Content.ReadAsStringAsync().Result
        Write-Log "ERROR: Intune assignment failed for '$userUPN' on '$deviceName'. $errorDetails"
        continue
    }
}

Disconnect-MgGraph
Write-Log "Disconnected from Graph. Script complete."
