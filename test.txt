# Import the GroupPolicy module
Import-Module GroupPolicy

#-----------------------------#
# CONFIGURATION SECTION
#-----------------------------#

$csvPath = "C:\Path\To\Your\file.csv"
$gpoName = "Win11 VDI Policy"
$context = "Computer"
$dryRun = $true

# Logging setup
$logDir = "$env:TEMP\GPORegistryLogs"
$logFile = Join-Path $logDir ("GPO_RegistryLog_{0:yyyy-MM-dd_HH-mm-ss}.log" -f (Get-Date))
New-Item -ItemType Directory -Path $logDir -Force | Out-Null

#-----------------------------#
# FUNCTION
#-----------------------------#

function Write-Log {
    param (
        [string]$Message,
        [string]$Level = "INFO"
    )
    $line = "[{0}] [{1}] {2}" -f (Get-Date -Format "yyyy-MM-dd HH:mm:ss"), $Level.ToUpper(), $Message
    Write-Host $line
    Add-Content -Path $logFile -Value $line
}

#-----------------------------#
# SCRIPT START
#-----------------------------#

Write-Log "Starting GPO registry import"
Write-Log "GPO: $gpoName | Context: $context | Dry Run: $dryRun"
Write-Log "CSV path: $csvPath"

# Load and validate CSV
if (-Not (Test-Path $csvPath)) {
    Write-Log "CSV file not found: $csvPath" -Level "ERROR"
    exit 1
}

$registryItems = Import-Csv -Path $csvPath

try {
    $gpo = Get-GPO -Name $gpoName -ErrorAction Stop
} catch {
    Write-Log "GPO '$gpoName' not found" -Level "ERROR"
    exit 1
}

foreach ($item in $registryItems) {
    $rootKey, $subKey = $item.Key -split "\\", 2
    $type = $item.Type.Trim()
    $name = $item.Name
    $rawValue = $item.Value

    $hiveMap = @{
        "HKLM" = "HKLM"
        "HKEY_LOCAL_MACHINE" = "HKLM"
        "HKCU" = "HKCU"
        "HKEY_CURRENT_USER" = "HKCU"
    }

    if (-not $hiveMap.ContainsKey($rootKey)) {
        Write-Log "Unsupported registry hive in key: $($item.Key)" -Level "WARN"
        continue
    }

    $hive = $hiveMap[$rootKey]
    $value = $null

    try {
        switch ($type) {
            "DWord"       { $value = [int]$rawValue }
            "QWord"       { $value = [long]$rawValue }
            "Binary"      { $value = $rawValue -split '-' | ForEach-Object { [byte]("0x$_") } }
            "MultiString" { $value = $rawValue -split '\|\|' }
            "String"      { $value = [string]$rawValue }
            "ExpandString"{ $value = [string]$rawValue }
            default {
                Write-Log "Unsupported registry type: $type" -Level "WARN"
                continue
            }
        }
    } catch {
        Write-Log "Type conversion failed for $($item.Key)\$name" -Level "ERROR"
        continue
    }

    Write-Log ("{0}: {1}\{2} = {3} ({4})" -f ($dryRun ? "DRY-RUN" : "SET"), "$hive\$subKey", $name, $value, $type)

    if (-not $dryRun) {
        try {
            Set-GPPrefRegistryValue -Name $gpoName `
                -Context $context `
                -Key "$hive\$subKey" `
                -ValueName $name `
                -Type $type `
                -Value $value `
                -Action Update

            Write-Log "Applied registry preference"
        } catch {
            Write-Log "Failed to apply registry item $($item.Key)\$name: $_" -Level "ERROR"
        }
    }
}

Write-Log "Finished processing registry items"
Write-Host "`nLog saved to: $logFile"
